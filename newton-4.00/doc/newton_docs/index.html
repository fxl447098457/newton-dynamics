Quick Reference

Include the main header to access the complete physics API:

#include <ndNewton.h>

// Create physics world
ndWorld world;
world.SetSubSteps(2);

// Create a dynamic body with sphere collision
ndSharedPtr<ndBody> body(new ndBodyDynamic());
ndShapeInstance sphere(new ndShapeSphere(1.0f));
body->SetCollisionShape(sphere);
body->SetMassMatrix(10.0f, sphere);
world.AddBody(body);

// Simulation loop
world.Update(1.0f / 60.0f);
world.Sync();
